  1. Sum and Average of 3 Numbers

    #include <stdio.h> int main() { // Sum and Average of 3 Numbers float num1, num2, num3; printf("Enter three numbers separated by spaces: "); scanf("%f %f %f", &num1, &num2, &num3); // Calculate sum float sum_of_numbers = num1 + num2 + num3; // Calculate average float average_of_numbers = sum_of_numbers / 3; // Display results printf("Sum of the numbers: %.2f\n", sum_of_numbers); printf("Average of the numbers: %.2f\n", average_of_numbers); return 0; }
